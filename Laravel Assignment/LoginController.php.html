use Laravel\Socialite\Facades\Socialite; use Illuminate\Support\Facades\Auth; use App\Models\User; public function redirectToGoogle() { return Socialite::driver('google')->redirect(); } public function handleGoogleCallback() { $googleUser = Socialite::driver('google')->stateless()->user(); $user = User::updateOrCreate([ 'email' => $googleUser->getEmail(), ], [ 'name' => $googleUser->getName(), 'google_id' => $googleUser->getId(), ]); Auth::login($user); return redirect('/calendar'); }
